// ========================================== // حصہ 2: سانپوں کا انجن، موومنٹ اور ٹچ کنٹرولز (VIP) // ========================================== const MAP_WIDTH = 4000; const MAP_HEIGHT = 4000; let player; let bots = []; let foods = []; let gameActive = false; let gameTick = 0; // موبائل ٹچ اور نائٹرو بوسٹر کنٹرولز let lastTap = 0; let isBoosting = false; class Snake { constructor(x, y, isBot, skinId, name) { this.x = x; this.y = y; this.isBot = isBot; this.skinId = skinId; this.name = name; this.radius = 16; this.baseSpeed = 3.5; this.speed = this.baseSpeed; this.angle = Math.random() * Math.PI * 2; this.targetAngle = this.angle; this.segments = []; this.length = 30; // VIP شروعاتی سائز this.turnSpeed = 0.09; // Worms Zone اسٹائل اسموتھ مڑنا // شروعاتی باڈی بنانا for (let i = 0; i < this.length; i++) { this.segments.push({ x: this.x, y: this.y }); } } update() { if (this.isBot) { this.handleAI(); } else { // اگر پلیئر ہے تو ڈبل ٹیپ پر اسپیڈ تیز کرنا this.speed = isBoosting ? this.baseSpeed * 1.8 : this.baseSpeed; // بوسٹ استعمال کرنے پر باڈی کا آہستہ آہستہ کم ہونا if (isBoosting && gameTick % 12 === 0 && this.length > 25) { this.length--; // یہاں مرے ہوئے سیگمنٹ کو کھانے میں بدلنے کی لاجک حصہ 4 میں آئے گی } } // زاویہ (Angle) کو اسموتھ طریقے سے مروڑنا let diff = this.targetAngle - this.angle; while (diff < -Math.PI) diff += Math.PI * 2; while (diff > Math.PI) diff -= Math.PI * 2; this.angle += diff * this.turnSpeed; // نئی پوزیشن پر لے جانا this.x += Math.cos(this.angle) * this.speed; this.y += Math.sin(this.angle) * this.speed; // میپ کی حدود کے اندر رکھنا this.x = Math.max(this.radius, Math.min(MAP_WIDTH - this.radius, this.x)); this.y = Math.max(this.radius, Math.min(MAP_HEIGHT - this.radius, this.y)); // باڈی کے ٹکڑوں کو سر کے پیچھے چلانا this.segments.unshift({ x: this.x, y: this.y }); while (this.segments.length > this.length) { this.segments.pop(); } // سائز کے حساب سے موٹائی (Radius) بڑھانا this.radius = 16 + Math.floor(this.length / 50); } handleAI() { // بوٹس کی اسمارٹ موومنٹ لاجک if (Math.random() < 0.02 || !this.closestFood) { let minDist = 500; this.closestFood = null; foods.forEach(f => { let d = Math.hypot(f.x - this.x, f.y - this.y); if (d < minDist) { minDist = d; this.closestFood = f; } }); } if (this.closestFood) { this.targetAngle = Math.atan2(this.closestFood.y - this.y, this.closestFood.x - this.x); } else if (Math.random() < 0.01) { // اگر کھانا نہ ملے تو رینڈم مڑنا this.targetAngle += (Math.random() - 0.5) * 2; } } draw(camX, camY) { ctx.save(); ctx.translate(camX, camY); // باڈی سیگمنٹس ڈرا کرنا for (let i = this.segments.length - 1; i >= 0; i--) { let seg = this.segments[i]; let fillStyle = '#ff9800'; if (this.skinId === 0) { fillStyle = (i % 6 < 3) ? '#ff9800' : '#ff3300'; // ٹائیگر پٹیاں } else if (this.skinId === 1) { fillStyle = (i % 6 < 3) ? '#00ffff' : '#0055ff'; // نیون نیلا } else if (this.skinId === 2) { fillStyle = (i % 8 < 4) ? '#ff007f' : '#000000'; // پنک اور بلیک زبرا } else { fillStyle = this.skinId; // بوٹس کے رنگ } ctx.fillStyle = fillStyle; ctx.beginPath(); ctx.arc(seg.x, seg.y, this.radius, 0, Math.PI * 2); ctx.fill(); } // سر اور چمکتی ہوئی آنکھیں (Worms Zone کی طرح فیس فیل) let head = this.segments[0]; let eyeOffset = this.radius - 5; // بائیں آنکھ let eX1 = head.x + Math.cos(this.angle + 0.4) * eyeOffset; let eY1 = head.y + Math.sin(this.angle + 0.4) * eyeOffset; ctx.fillStyle = '#fff'; ctx.beginPath(); ctx.arc(eX1, eY1, 5, 0, Math.PI * 2); ctx.fill(); ctx.fillStyle = '#000'; ctx.beginPath(); ctx.arc(eX1, eY1, 2.5, 0, Math.PI * 2); ctx.fill(); // دائیں آنکھ let eX2 = head.x + Math.cos(this.angle - 0.4) * eyeOffset; let eY2 = head.y + Math.sin(this.angle - 0.4) * eyeOffset; ctx.fillStyle = '#fff'; ctx.beginPath(); ctx.arc(eX2, eY2, 5, 0, Math.PI * 2); ctx.fill(); ctx.fillStyle = '#000'; ctx.beginPath(); ctx.arc(eX2, eY2, 2.5, 0, Math.PI * 2); ctx.fill(); // کیڑے کے اوپر نام دکھانا ctx.fillStyle = 'rgba(255,255,255,0.8)'; ctx.font = 'bold 13px Arial'; ctx.textAlign = 'center'; ctx.fillText(this.name, head.x, head.y - this.radius - 10); ctx.restore(); } } // موبائل ٹچ ایونٹس انجن window.addEventListener('touchstart', (e) => { if (!gameActive) return; let now = Date.now(); // اگر 250 میلی سیکنڈ کے اندر دوبارہ ٹیپ ہو تو اسپیڈ بوسٹ آن if (now - lastTap < 250) { isBoosting = true; } // چمکتا ہوا گراؤنڈ گلو ایفیکٹ ctx.shadowBlur = 10; ctx.shadowColor = f.color; ctx.fillStyle = f.color; ctx.beginPath(); ctx.arc(f.x, f.y, f.radius, 0, Math.PI * 2); ctx.fill(); } ctx.restore(); } // میپ کی جادوئی نیٹ گریڈ لائنز ڈرا کرنا (Worms Zone ارینا فیل) MapGridBackground = function(camX, camY) { ctx.save(); ctx.strokeStyle = '#162238'; ctx.lineWidth = 1.5; let gridSize = 160; let startX = camX % gridSize; let startY = camY % gridSize; for (let x = startX; x < canvas.width; x += gridSize) { ctx.beginPath(); ctx.lineWidth = 8; ctx.strokeRect(camX, camY, MAP_WIDTH, MAP_HEIGHT); ctx.restore(); }; // پرو مینی میپ ریڈار (Mini-Map Radar) function drawMiniMap() { let size = 130; // نقشے کا سائز let margin = 15; bots.forEach(b => { let bx = mX + (b.x * ratio); let by = mY + (b.y * ratio); ctx.beginPath(); ctx.arc(bx, by, 1.5, 0, Math.PI * 2); ctx.fill(); }); // نقشے پر اصلی پلیئر (Asim) کو چمکتے ہوئے نیون نقطے سے دکھانا if (player) { ctx.fillStyle = '#00ff00'; ctx.shadowBlur = 6; ctx.shadowColor = '#00ff00'; let px = mX + (player.x * ratio); let py = mY + (player.y * ratio); ctx.beginPath(); ctx.arc(px, py, 3, 0, Math.PI * 2); ctx.fill(); ctx.shadowBlur = 0; // ری سیٹ } } // ========================================== // حصہ 4: ٹکراؤ (Collisions)، لیڈر بورڈ اور فائنل گیم لوپ (VIP) // ========================================== // گیم شروع کرنے کا مین فنکشن function startGame() { document.getElementById('mainMenu').style.display = 'none'; document.getElementById('topUI').style.display = 'flex'; document.getElementById('leaderboard').style.display = 'block'; gameTick = 0; playerData.score = 0; playerData.level = 1; isBoosting = false; // پلیئر کو میپ کے سینٹر میں سپون کرنا player = new Snake(MAP_WIDTH / 2, MAP_HEIGHT / 2, false, playerData.currentSkin, "ASIM GAMES"); // 10 اسمارٹ اے آئی بوٹس بنانا bots = []; let botNames = ["WormKing", "TurboSlayer", "AlphaWorm", "Shadow", "Speedy", "AsimBot", "Glutton", "Matrix", "Phantom", "Apex"]; let botColors = ['#e91e63', '#9c27b0', '#2196f3', '#4caf50', '#009688', '#ffeb3b', '#ff5722', '#00e676', '#00b0ff', '#ff3d00'];
